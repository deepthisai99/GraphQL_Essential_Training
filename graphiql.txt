# Querying with fragments
query{
  widget1: getProduct(id: "640630dbe24b3d6388be9b41"){
    ...widgetFragment
  }
  widget2: getProduct(id: "64073c927e8d39fd3fc3c8ed"){
  ...widgetFragment
  }
}

fragment widgetFragment on Product{
  id
  name
  description
  price
  soldout
  inventory
}
